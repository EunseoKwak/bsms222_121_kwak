inside1 <- replicate(B1, {
  x <- sample(c(0,1), size = N1, replace = TRUE, prob = c(1-p, p))
  x_hat <- mean(x)
  se_hat <- sqrt(x_hat * (1 - x_hat) / N)
  between(p, x_hat - 1.96 * se_hat, x_hat + 1.96 * se_hat)
  return(c(x_hat, se_hat))
})
